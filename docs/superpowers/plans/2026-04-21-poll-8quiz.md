# Poll And 8-Question Quiz Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Restore execution for poll-shaped daily activities and 8-question browser quizzes without regressing the current standard quiz flow.

**Architecture:** Keep the existing API-driven quiz path for standard 20/30/40/50-point quizzes, but add browser-driven handling for the two currently missing cases: poll promotions and 8-question quiz pages. Route poll detection in `Workers`, expose `Poll` and page-aware `Quiz` entrypoints in `Activities`, and keep page-interaction details isolated in focused browser activity code.

**Tech Stack:** TypeScript, Patchright `Page`, Node.js `node:test`, existing bot logger/browser utility layer, `npm run build`

```ts
import type { Page } from 'patchright'
import type { BasePromotion } from '../../../interface/DashboardData'
import { Workers } from '../../Workers'

const POLL_OPTION_SELECTORS = [
    'input[type="radio"]',
    '.btOption',
    '.rqOption',
    '[role="radio"]'
]

export class Poll extends Workers {
    public async doPoll(promotion: BasePromotion, page: Page) {
        const oldBalance = Number(this.bot.userData.currentPoints ?? 0)

        for (const selector of POLL_OPTION_SELECTORS) {
            const count = await page.locator(selector).count().catch(() => 0)
            if (!count) continue

            const clicked = await this.bot.browser.utils.ghostClick(page, `${selector}:nth-of-type(1)`)
            if (!clicked) continue

            await this.bot.utils.wait(this.bot.utils.randomDelay(1500, 3000))

            const newBalance = await this.bot.browser.func.getCurrentPoints().catch(() => oldBalance)
            const gained = Math.max(0, newBalance - oldBalance)

            if (gained > 0) {
                this.bot.userData.currentPoints = newBalance
                this.bot.userData.gainedPoints = (this.bot.userData.gainedPoints ?? 0) + gained
            }

            return
        }

        this.bot.logger.warn(this.bot.isMobile, 'POLL', `未找到可点击投票选项 | offerId=${promotion.offerId}`)
    }
}
```

```ts
import type { Page } from 'patchright'

async doQuiz(promotion: BasePromotion, page?: Page) {
    // existing setup...

    if (promotion.activityProgressMax === 80) {
        if (!page) {
            this.bot.logger.warn(
                this.bot.isMobile,
                'QUIZ',
                `8题测验缺少页面上下文，无法执行 | offerId=${offerId}`
            )
            return
        }

        await this.runEightQuestionQuiz(promotion, page, startBalance)
        return
    }

    // existing standard ReportActivity path...
}

async runEightQuestionQuiz(promotion: BasePromotion, page: Page, startBalance: number) {
    const answerSelectors = ['input[type="radio"]', '.rqOption', '.btOption', '[role="option"]']
    let progressStalls = 0

    while (progressStalls < 3) {
        const completed = await page.locator('#quizCompleteContainer').count().catch(() => 0)
        if (completed) break

        let clicked = false
        for (const selector of answerSelectors) {
            const count = await page.locator(selector).count().catch(() => 0)
            if (!count) continue

            clicked = await this.bot.browser.utils.ghostClick(page, `${selector}:nth-of-type(1)`)
            if (clicked) break
        }

        if (!clicked) {
            progressStalls++
            await this.bot.utils.wait(this.bot.utils.randomDelay(1500, 2500))
            continue
        }

        await this.bot.utils.wait(this.bot.utils.randomDelay(2000, 3500))
    }

    const newBalance = await this.bot.browser.func.getCurrentPoints().catch(() => startBalance)
    const gained = Math.max(0, newBalance - startBalance)
    this.bot.userData.currentPoints = newBalance
    this.bot.userData.gainedPoints = (this.bot.userData.gainedPoints ?? 0) + gained
}
```
